from django.db import transaction
from django.contrib.auth import get_user_model

from apps.core.models import TenantMembership
from .models import Parent


def ensure_parent_for_student(
    *,
    tenant,
    parent_phone: str,
    student_name: str,
    parent_password: str,
) -> Parent:
    """
    í•™ë¶€ëª¨ ì „í™”ë²ˆí˜¸ë¡œ Parent ì¡°íšŒ ë˜ëŠ” ìƒì„±
    - ì—†ìœ¼ë©´ User + Parent + TenantMembership ìƒì„±
    - ìˆìœ¼ë©´ ê¸°ì¡´ Parent ë°˜í™˜ (User ì—†ìœ¼ë©´ ìƒì„±)
    """
    parent_phone = str(parent_phone or "").strip()
    if not parent_phone:
        raise ValueError("í•™ë¶€ëª¨ ì „í™”ë²ˆí˜¸ëŠ” í•„ìˆ˜ì…ë‹ˆë‹¤.")

    User = get_user_model()
    # tenant ë‚´ ìœ ì¼í•œ í•™ë¶€ëª¨ ì‹ë³„: username = p_{tenant_id}_{phone}
    parent_username = f"p_{tenant.id}_{parent_phone}"

    parent = Parent.objects.filter(tenant=tenant, phone=parent_phone).first()

    if parent:
        if not parent.user_id:
            with transaction.atomic():
                user = User.objects.create_user(
                    username=parent_username,
                    phone=parent_phone,
                    name=f"{student_name} í•™ë¶€ëª¨",
                )
                user.set_password(parent_password)
                user.save()
                parent.user = user
                parent.save(update_fields=["user"])
                TenantMembership.ensure_active(
                    tenant=tenant,
                    user=user,
                    role="parent",
                )
        return parent

    with transaction.atomic():
        if User.objects.filter(username=parent_username).exists():
            raise ValueError(f"í•™ë¶€ëª¨ ì „í™”ë²ˆí˜¸ {parent_phone}ê°€ ì´ë¯¸ ë‹¤ë¥¸ í•™ì›ì—ì„œ ì‚¬ìš© ì¤‘ì…ë‹ˆë‹¤.")

        user = User.objects.create_user(
            username=parent_username,
            phone=parent_phone,
            name=f"{student_name} í•™ë¶€ëª¨",
        )
        user.set_password(parent_password)
        user.save()

        parent = Parent.objects.create(
            tenant=tenant,
            user=user,
            name=f"{student_name} í•™ë¶€ëª¨",
            phone=parent_phone,
        )

        TenantMembership.ensure_active(
            tenant=tenant,
            user=user,
            role="parent",
        )

    return parent
