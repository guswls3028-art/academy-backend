class StudentListSerializer(serializers.ModelSerializer):
    tags = TagSerializer(many=True, read_only=True)
    enrollments = EnrollmentSerializer(many=True, read_only=True)
    is_enrolled = serializers.SerializerMethodField()
    profile_photo_url = serializers.SerializerMethodField()

    class Meta:
        model = Student
        fields = "__all__"
        ref_name = "StudentList"

    def get_profile_photo_url(self, obj):
        if not obj.profile_photo:
            return None
        request = self.context.get("request")
        if request:
            return request.build_absolute_uri(obj.profile_photo.url)
        return obj.profile_photo.url

    def to_representation(self, obj):
        data = super().to_representation(obj)
        data["profile_photo_url"] = self.get_profile_photo_url(obj)
        return data

    def get_is_enrolled(self, obj):
        request = self.context.get("request")
        if not request:
            return False

        lecture_id = request.query_params.get("lecture")
        if lecture_id:
            try:
                lid = int(lecture_id)
            except (TypeError, ValueError):
                return False
            return obj.enrollments.filter(lecture_id=lid).exists()

        return False

    @transaction.atomic
    def create(self, request, *args, **kwargs):
        """
        í•™ìƒ ìƒì„± ì‹œ ì²˜ë¦¬ íë¦„

        1. ì…ë ¥ê°’ ê²€ì¦ (StudentCreateSerializer)
        2. í•™ë¶€ëª¨ ê³„ì • ìƒì„±/ì—°ê²° (ensure_parent_for_student)
        3. User ìƒì„± (username = ps_number)
        4. Student ìƒì„± + tenant / user / parent ì—°ê²°
        5. TenantMembership(role=student) SSOT ê°•ì œ ìƒì„±
        6. (ì˜µì…˜) ê°€ì… ì„±ê³µ ë©”ì‹œì§€ ì¼ê´„ ë°œì†¡
        """
        serializer = self.get_serializer(
            data=request.data,
            context={"request": request},
        )
        serializer.is_valid(raise_exception=True)

        User = get_user_model()
        data = serializer.validated_data
        send_welcome = data.pop("send_welcome_message", False)

        phone = data.get("phone")  # nullable
        password = data.pop("initial_password")
        parent_phone = data.get("parent_phone", "")
        ps_number = data.get("ps_number")

        # 1ï¸âƒ£ í•™ë¶€ëª¨ ê³„ì • ìƒì„± (ID = í•™ë¶€ëª¨ ì „í™”ë²ˆí˜¸)
        parent = None
        if parent_phone:
            parent = ensure_parent_for_student(
                tenant=request.tenant,
                parent_phone=parent_phone,
                student_name=data.get("name", ""),
                parent_password=password,
            )

        # 2ï¸âƒ£ User ìƒì„± (phoneì´ ì—†ìœ¼ë©´ usernameë§Œ ì‚¬ìš©)
        user = student_repo.user_create_user(
            username=ps_number,
            phone=phone or "",
            name=data.get("name", ""),
        )
        user.set_password(password)
        user.save()

        # 3ï¸âƒ£ Student ìƒì„± + parent ì—°ê²°
        student = student_repo.student_create(
            tenant=request.tenant,
            user=user,
            parent=parent,
            **data,
        )

        # 4ï¸âƒ£ TenantMembership
        TenantMembership.ensure_active(
            tenant=request.tenant,
            user=user,
            role="student",
        )

        # 5ï¸âƒ£ ê°€ì… ì„±ê³µ ë©”ì‹œì§€ ë°œì†¡
        if send_welcome:
            site_url = get_site_url(request)
            send_welcome_messages(
                created_students=[student],
                student_password=password,
                parent_password_by_phone={parent_phone: password} if parent_phone else {},
                site_url=site_url,
            )

        output = StudentDetailSerializer(
            student,
            context={"request": request},
        )
        return Response(output.data, status=201)

    deleted_student = student_repo.student_filter_tenant_name_parent_phone_deleted(

        school_val = (item.get("school") or "").strip() or None

from typing import Tuple

# HIGH / MIDDLE
SchoolType = str


def normalize_school_from_name(
    school_name: str | None,
    school_type_from_input: str | None = None,
) -> Tuple[SchoolType, str | None, str | None]:
    """
    í•™êµëª…ê³¼ (ì„ íƒ) ì…ë ¥ëœ í•™êµ íƒ€ì…ìœ¼ë¡œ (school_type, high_school, middle_school) ë°˜í™˜.

    - í•™êµëª… ë¹„ì–´ìˆìŒ â†’ high_school/middle_school ì€ None, school_type ì€ ì…ë ¥ê°’ ë˜ëŠ” "HIGH".

    - í•™êµëª…ì— "ì¤‘" í¬í•¨(ì˜ˆ: íœ˜ë¬¸ì¤‘, XXì¤‘) â†’ MIDDLE, middle_school=í•™êµëª…, high_school=None.
    - ë‘˜ ë‹¤ í¬í•¨(ì˜ˆ: ì¤‘ë™ê³ ) â†’ "ê³ " ìš°ì„  ì ìš©í•˜ì—¬ HIGH.
    - ê·¸ ì™¸ â†’ school_type_from_input ë˜ëŠ” "HIGH" ë¡œ high_school/middle_school ë°°ì¹˜.
    """
    name = (school_name or "").strip() or None
    default_type = (school_type_from_input or "HIGH").upper()
    if default_type not in ("HIGH", "MIDDLE"):
        default_type = "HIGH"

    if not name:
        return (default_type, None, None)

    # "ê³ "ê°€ ìˆìœ¼ë©´ ê³ ë“±(ì¤‘ë™ê³ , XXê³ ë“±í•™êµ ë“±)
    if "ê³ " in name:
        return ("HIGH", name, None)
    if "ì¤‘" in name:
        return ("MIDDLE", None, name)

    # êµ¬ë¶„ ë¶ˆê°€ ì‹œ ì…ë ¥ íƒ€ì…ìœ¼ë¡œ
    if default_type == "HIGH":
        return ("HIGH", name, None)
    return ("MIDDLE", None, name)
